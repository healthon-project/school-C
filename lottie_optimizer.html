(function() {
  const wrap = document.getElementById('lottie-entertainment');
  if (!wrap) return;

  let loaded = false;

  function loadAnim() {
    if (loaded) return;
    loaded = true;
    lottie.loadAnimation({
      container: wrap,
      renderer: 'svg',
      loop: true,
      autoplay: true,
      animationData: LOTTIE_DATA  // ← path 대신 animationData 사용
    });
    // 로드 후 메모리 해제
    LOTTIE_DATA = null;
  }

  // 화면에 보일 때만 로드 (IntersectionObserver)
  if ('IntersectionObserver' in window) {
    const obs = new IntersectionObserver((entries) => {
      if (entries[0].isIntersecting) {
        loadAnim();
        obs.disconnect();  // 한 번 로드 후 옵저버 해제
      }
    }, { rootMargin: '100px' });  // 100px 전에 미리 로드
    obs.observe(wrap);
  } else {
    setTimeout(loadAnim, 500);  // 폴백: 0.5초 후 로드
  }
})();
다중 로티 파일 매니저
// ─── 로티 데이터 레지스트리 ───────────────────────
const LOTTIE_REGISTRY = {
  'entertainment': { data: /* entertainment.json */, loaded: false, anim: null },
  'cat-rocket':    { data: /* cat_rocket.json */,    loaded: false, anim: null },
  'bulldog':       { data: /* bulldog_rocket.json */, loaded: false, anim: null },
};

// ─── 로티 매니저 클래스 ───────────────────────────
class LottieManager {
  constructor() {
    this.registry = LOTTIE_REGISTRY;
    this.observer = null;
    this._initObserver();
  }

  _initObserver() {
    this.observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const key = entry.target.getAttribute('data-lottie');
          this.load(key, entry.target);
          this.observer.unobserve(entry.target);
        }
      });
    }, { rootMargin: '120px', threshold: 0 });
  }

  register(el) {
    // HTML: <div data-lottie="entertainment"></div>
    const key = el.getAttribute('data-lottie');
    if (this.registry[key]) {
      this.observer.observe(el);
    }
  }

  load(key, container) {
    const entry = this.registry[key];
    if (!entry || entry.loaded) return;
    entry.loaded = true;
    entry.anim = lottie.loadAnimation({
      container,
      renderer: 'svg',
      loop: true,
      autoplay: true,
      animationData: entry.data
    });
    entry.data = null;  // 메모리 해제
  }

  pause(key)   { this.registry[key]?.anim?.pause(); }
  play(key)    { this.registry[key]?.anim?.play(); }
  destroy(key) { this.registry[key]?.anim?.destroy(); }

  // 탭 전환 시: 보이지 않는 로티 일시정지
  pauseAll() {
    Object.values(this.registry).forEach(e => e.anim?.pause());
  }
}

// ─── 초기화 ──────────────────────────────────────
const lottieManager = new LottieManager();
document.querySelectorAll('[data-lottie]').forEach(el => {
  lottieManager.register(el);
});
React — useLottie Hook
// hooks/useLottie.js
import { useEffect, useRef, useState } from 'react';
import lottie from 'lottie-web';

export function useLazyLottie(animationData, options = {}) {
  const containerRef = useRef(null);
  const animRef = useRef(null);
  const [status, setStatus] = useState('idle'); // idle | loading | ready

  useEffect(() => {
    const el = containerRef.current;
    if (!el) return;

    const observer = new IntersectionObserver(([entry]) => {
      if (!entry.isIntersecting) return;
      observer.disconnect();
      setStatus('loading');

      // requestIdleCallback: 브라우저 유휴 시간에 로드
      (window.requestIdleCallback || window.setTimeout)(() => {
        animRef.current = lottie.loadAnimation({
          container: el,
          renderer: 'svg',
          loop: options.loop ?? true,
          autoplay: options.autoplay ?? true,
          animationData,
        });
        setStatus('ready');
      });
    }, { rootMargin: '150px' });

    observer.observe(el);
    return () => {
      observer.disconnect();
      animRef.current?.destroy(); // 언마운트 시 정리
    };
  }, []);

  return { containerRef, status, anim: animRef };
}
